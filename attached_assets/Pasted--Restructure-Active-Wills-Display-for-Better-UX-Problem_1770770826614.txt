// Organize Wills into sections

const soloWills = activeWills.filter(will => !will.circle_id);
const circleWills = activeWills.filter(will => will.circle_id);

return (
  <>
    <Header title="My Wills" backButton />
    
    {soloWills.length > 0 && (
      <section>
        <h2>Solo Wills ({soloWills.length})</h2>
        {soloWills.map(will => (
          <WillCard key={will.id} will={will} />
        ))}
      </section>
    )}
    
    {circleWills.length > 0 && (
      <section>
        <h2>Circle Wills ({circleWills.length})</h2>
        {circleWills.map(will => (
          <WillCard key={will.id} will={will} />
        ))}
      </section>
    )}
  </>
);
