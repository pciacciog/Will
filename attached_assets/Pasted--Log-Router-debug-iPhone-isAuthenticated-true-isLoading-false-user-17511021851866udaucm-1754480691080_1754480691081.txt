    const authResponse = await fetch('/api/auth/me');
    if (!authResponse.ok) {
      console.log('❌ Not logged in - please sign in first');
      return null;
    }

    console.log('✅ User logged in:', user.firstName, user.lastName);

    return null;
  }
}
